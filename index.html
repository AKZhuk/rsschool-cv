class ArticlesController < ApplicationControl
    def new
        @article = Article.new

    def edit
        @article = Article.find(params[:id])

    def create
          @query = Query.new(params.require(:article)
            .permit(:name, :email, :organization, :text))
          @query.save
          redirect_to articles_path

    def update
        @article = Article.find(params[:id])
          if @article.update(params.require(:article).
            permit(:title, :text))
            redirect_to @article
          else
            render 'edit'
          end

    def show
        @article = Article.find(params[:id])

    def index
        @articles = Article.all

    def destroy
        @article = Article.find(params[:id])
        @article.destroy
        redirect_to articles_path
    end
end
